Click Here

Attaches a jQuery UI dialog box to the "Click here" link.

This page is ~59% faster than the standard implementation. Waterfall Chart (this page is on the bottom) 

<script type="text/javascript" src="js/Importer.js">
  var clickLink = document.getElementById("click-here");
  var linkAction = function(){
    JS.Use("jquery",function(){
      JS.Require("/js/jqueryui.js",true,function(){
        $("body").append("<div id='dialog'>Change the menu color?</div>");
        $("#dialog").dialog({
          autoOpen: false,
          buttons:{
            Yes: function(){
              Links.change();
              $("#dialog").dialog("close");
            },
            No: function(){
              $("#dialog").dialog("close");
            }
          }
        });
        JS.Use("evaltest.links",function(){
          Links.addEvent("#click-here");
          clickLink.removeEventListener("mouseover",linkAction,false);            
        });
      });
    });
  };
  JS.Require("js/libraries/test-lib.jstxt",false,function(){
    clickLink.addEventListener("mouseover",linkAction,false);
  });
</script>
