<script type="text/javascript" src="js/Importer.js">
      $("#click-here").bind("mouseover",function(){
        $("body").append("<div id='dialog'>Change the menu color?</div>");
        $("#dialog").dialog({
          autoOpen: false,
          buttons:{
            Yes: function(){
              Links.change();
              $("#dialog").dialog("close");
            },
            No: function(){
              $("#dialog").dialog("close");
            }
          }
        });
        Links.addEvent("#click-here");
        Menu.create("menu");
        $("#click-here").unbind("mouseover");
      });
</script>
