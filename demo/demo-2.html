Click Here

Attaches a jQuery UI dialog box to the "Click here" link.

Waterfall Chart (this page is on the top) 

<script type="text/javascript">
      $("#click-here").bind("mouseover",function(){
        $("body").append("<div id='dialog'>Change the menu color?<div>");
        $("#dialog").dialog({
          autoOpen: false,
          buttons:{
            Yes: function(){
              Links.change();
              $("#dialog").dialog("close");
            },
            No: function(){
              $("#dialog").dialog("close");
            }
          }
        });
        Links.addEvent("#click-here");
        $("#click-here").unbind("mouseover");
      });
</script>
