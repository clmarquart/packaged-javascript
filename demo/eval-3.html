Use It!

This page is ~56% faster than the standard implementation. Waterfall Chart (this page is on the bottom) 

<script type="text/javascript" src="js/Importer.js">
      JS.Require("js/libraries/test-lib.jstxt");
      
      var useItLink = document.getElementById("click-here"); 
      var useIt = function(){
        JS.Use("jquery")
          .Use("evaltest.menu",function(){
             Menu.create("menu")
          })
          .Require("/js/jqueryui.js",true,function(){
            $("body").append("<div id='dialog'>Change the menu color?<div>");
            $("#dialog").dialog({
              autoOpen: false,
              buttons:{
                Yes: function(){
                  Links.change();
                  $("#dialog").dialog("close");
                },
                No: function(){
                  $("#dialog").dialog("close");
                }
              }
            });
            JS.Use("evaltest.links",function(){
              Links.addEvent("#click-here");            
            });
          });
        useItLink.removeEventListener("mouseover",useIt,false);
      }
      useItLink.addEventListener("mouseover",useIt,false);
</script>
