Create a click activated drop-down menu.

This page is ~29% faster than the standard implementation. Waterfall Chart (this page is on the bottom) 

<script type="text/javascript" src="js/Importer.js">
  JS.Require("js/libraries/test-lib.jstxt");

  var themenu = document.getElementById("menu");
  var createMenu = function(){
        JS.Use("jquery",function(){
          JS.Use("evaltest.menu",function(){
            Menu.create("menu");
            themenu.removeEventListener("mouseover",createMenu,false);
          });
        });
    };
  themenu.addEventListener("mouseover",createMenu,false);
</script>
